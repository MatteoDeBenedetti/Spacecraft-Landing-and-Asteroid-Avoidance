#------------------------------------------------------------------------------------
# Target file to be compiled
#------------------------------------------------------------------------------------
MAIN = lem

#------------------------------------------------------------------------------------
# Compiler to be used
#------------------------------------------------------------------------------------
CC = gcc

#------------------------------------------------------------------------------------
# Options
#------------------------------------------------------------------------------------
CFLAGS = -Wall -lrt -lm -pthread

#------------------------------------------------------------------------------------
# Object files to be linked
#------------------------------------------------------------------------------------
OBJ1 = pthread_lib
OBJ2 = allegro_lib
OBJ3 = space_lib

OBJS = ${MAIN}.o ${OBJ1}.o ${OBJ2}.o ${OBJ3}.o

#------------------------------------------------------------------------------------
# External libraries
#------------------------------------------------------------------------------------
LIBS = `allegro-config --libs`

#------------------------------------------------------------------------------------
# Dependencies
#------------------------------------------------------------------------------------
${MAIN}: ${OBJS}
	${CC} -o ${MAIN} ${OBJS} ${LIBS} ${CFLAGS}
	
${MAIN}.o: ${MAIN}.c
	${CC} -c ${MAIN}.c

${OBJ1}.o: ${OBJ1}.c
	${CC} -c ${OBJ1}.c	
	
${OBJ2}.o: ${OBJ2}.c
	${CC} -c ${OBJ2}.c	
	
${OBJ3}.o: ${OBJ3}.c
	${CC} -c ${OBJ3}.c
